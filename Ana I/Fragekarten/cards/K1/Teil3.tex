\card{Was ist der Existenzquantor}{20}{
	Für eine Menge $A$ und eine Aussageform $a(x)$ besagt der Existenzquantor, dass \gqq{es gibt ein $x$ in $A$, sodass $a(x)$ erfüllt ist} $\equiv \exists x\in A : a(x)$.\\
	Für \gqq{es gibt genau ein $x$, für das $a(x)$ gilt} schreibt man $\exists! x\in A : a(x)$.
}
\card{Was ist der Allquantor}{20}{
	Für eine Menge $A$ und eine Aussageform $a(x)$ besagt der Allquantor, dass \gqq{Für alle $x\in A$ gilt $a(x)$}$\equiv\forall x\in A:a(x)$.
}
