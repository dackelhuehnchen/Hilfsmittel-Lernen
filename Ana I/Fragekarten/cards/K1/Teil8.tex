\card{Wie sind die ganzen Zahlen definiert}{109}{
	$\Z:=\{n-m:n,m\in\N\}$
}
\card{Wie sind die rationalen Zahlen definiert}{109}{
	$\Q:=\left\{\dfrac{p}{q}:p,q\in\Z\und q\neq 0\right\}$
}
\card{Wie sind die irrationalen Zahlen definiert}{109}{
	$\I:=\R\setminus \Q$
}
\card{Wie sind die komplexen Zahlen definiert}{109}{
	$\C:=\{(a,b):a,b\in \R\}$ mit $(a,b)=a+ib$ und\\
	$(a,b)+(c,d)=(a+c,b+d)$ sowie\\
	$(a,b)\cdot (c,d)=(ac-bd,bc+ad)$
}
