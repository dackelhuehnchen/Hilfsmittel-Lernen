\card{Wann ist eine Abbildung $f:E\rightarrow F$ metrischer Räume $\varepsilon$-$\delta$-stetig in $x_0\in E$}{25}{
	$\underset{\varepsilon>0}{\forall}\underset{\delta>0}{\exists}\underset{x\in E}{\forall} d(x,x_0)<\delta \Rightarrow d(f(x),f(x_0))<\varepsilon\\
	\hspace*{1cm}\equiv\underset{\varepsilon>0}{\forall}\underset{\delta>0}{\exists}f(\ball(\delta,x_0))\subset \ball(\varepsilon,f(x_0))$
}
\card{Wann ist eine Abbildung $f:E\rightarrow F$ metrischer Räume $\varepsilon$-$\delta$-stetig}{25}{
	Wenn $f$ in allen Punkten $\varepsilon$-$\delta$-stetig ist.
}
\card{Wann ist eine Abbildung $f:E\rightarrow F$ metrischer Räume als topologische Abbildung in $x_0\in E$ stetig}{25}{
	$\underset{V\in\mathcal{U}(f(x_0))}{\forall}~\underset{U\in\mathcal{U}(x_0)}{\exists}~f(U)\subset V\\
	\hspace*{1cm}\equiv\underset{V\in\mathcal{U}(f(x_0))}{\forall}~f^{-1}(V)\in \mathcal{U}(x_0)$
}
\card{Wann ist eine Abbildung $f:E\rightarrow F$ metrischer Räume als topologische Abbildung stetig}{25}{
	Wenn $f$ in allen Punkten als topologische Abbildung stetig ist.
}
\card{Wann ist eine Abbildung $f:E\rightarrow F$ metrischer Räume folgenstetig in $x_0\in E$}{25}{
	Falls für alle Folgen $\folge(x;n;\N)\subset E$ mit $x_n\rightarrow x_0$ auch $f(x_n)\rightarrow f(x_0)$ für $n\rightarrow \infty$ folgt.
}
\card{Wann ist eine Abbildung $f:E\rightarrow F$ metrischer Räume in $x_0\in E$ stetig}{25}{
	Falls $f$ in $x_0$ $\varepsilon$-$\delta$-stetig, als topologischer Abbildung oder folgenstetig ist.\\
	Oder, falls $f^{-1}(X)$ abgeschlossen ist, für alle\\
	$X\subset F$, die abgeschlossen sind.
}
\card{Wann ist eine Abbildung $f:E\rightarrow F$ metrischer Räume stetig}{25}{
	Falls $f$ in allen Punkten stetig ist.
}

	$\underset{\varepsilon>0}{\forall}~\underset{\delta>0}{\exists}~\underset{x,y\in E}{\forall}d(x,y)<\delta \Rightarrow d(f(x),f(y))<\varepsilon$
