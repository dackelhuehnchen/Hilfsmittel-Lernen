\begin{algorithm}[H]
	\SetAlgoVlined
	\SetKwProg{Fn}{Function}{}{end}
	\KwIn{Pattern $P[1,\dots,m]$}
	\KwOut{\bound~$\pi$}
	\BlankLine
	\Begin{
		$\pi(1) \leftarrow 0$\\
		$q\leftarrow 0$\\
		\For{$i\leftarrow 2,\dots,m$}{
			\While{$q>0$ und $P[q+1]\neq P[i]$}{
				$q\leftarrow \pi(q)$
			}
			\If{$P[q+1]=T[i]$}{
				$q\leftarrow q1$
			}
			$\pi(i) \leftarrow q$
		}
		\KwRet{$\pi$}
	}
\end{algorithm}
